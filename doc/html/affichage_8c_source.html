55 void affichage(board_t *self) {

57  typedef line board[self->dims.h];
58 
59  printf("\033[2J");
60  fflush(stdout);

62  printf("%*s\r\n", self->dims.w + 3, "🤔");
63  //
64  board *b = (void*)self->board;
65 
66  const char *target = " ";
67  for (int y = 0; y < self->dims.h; ++y) {
68  for (int x = 0; x < self->dims.w; ++x) {
69  cell_t c = (*b)[y][x];
70  int r = !!(c.active);
71  r += !!(c.flagged) * 2;
72  if (c.bombe)
73  target = (const char *[]) {"＃", 0, "🏳 " }[r];
74  else {
75  target = (const char *[]) { "＃", digits[c.n], "🏳 " }[r];
76  }
77  if (self->cursor.x == x &&
78  self->cursor.y == y)
79  printf("\033[7m%s\033[0m", target);
